1629 template<typename L, typename T>
1630 class ConstListIterator {
1631  public:
1632  ConstListIterator<L,T>(const L& list, int index) : m_list(list), m_index(index) {}
1633 
1634  const T operator*() const { return m_list[m_index]; }
1635  const ConstListIterator<L,T>& operator++() { ++m_index; return *this; }
1636  bool operator!=(const ConstListIterator<L,T>& rhs) const { return m_index != rhs.m_index; }
1637 
1638  private:
1639  const L& m_list;
1640  int m_index;
1641 };
